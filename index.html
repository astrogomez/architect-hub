       function init() {

           const container = document.getElementById('day-selector-container');

       function setDay(day) {

       function togglePlay() {

           if (audio.paused) {

               audio.play();

               playIcon.innerHTML = '<path d="M6 19h4V5H6v14zm8-14v14h4V5h-4z"/>';

               playBtn.classList.remove('pulsing');

           } else {

               audio.pause();

               playIcon.innerHTML = '<path d="M8 5v14l11-7z"/>';

               playBtn.classList.add('pulsing');

           }

       }

       function seek(e) {

           const rect = e.currentTarget.getBoundingClientRect();

           const x = e.clientX - rect.left;

           const width = rect.width;

           audio.currentTime = (x / width) * audio.duration;

       }

       audio.ontimeupdate = function() {

           if (audio.duration) {

               const perc = (audio.currentTime / audio.duration) * 100;

               progress.style.width = perc + '%';

             

               const curM = Math.floor(audio.currentTime / 60);

               const curS = Math.floor(audio.currentTime % 60);

               timeDisplay.innerText = `${curM}:${curS < 10 ? '0' : ''}${curS}`;

             

               const durM = Math.floor(audio.duration / 60);

               const durS = Math.floor(audio.duration % 60);

               durationDisplay.innerText = `${durM}:${durS < 10 ? '0' : ''}${durS}`;

           }

       }
